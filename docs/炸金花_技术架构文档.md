## 3. 数据库设计

### 3.1 MongoDB Collections

**users（用户表）**
```javascript
{
  _id: ObjectId,
  openid: String,           // 微信openid（单人版用设备ID）
  nickname: String,          // 昵称
  avatar: String,            // 头像URL
 金币: Number,              // 金币数量
  房卡: Number,              // 房卡数量
  createdAt: Date,
  updatedAt: Date
}
```

**match_records（战绩表）**
```javascript
{
  _id: ObjectId,
  userId: ObjectId,         // 用户ID
  roomId: String,            // 房间ID
  牌型: String,             // 最终牌型
  排名: Number,             // 第几名
  获得金币: Number,          // 赢了/输了多少
  耗时: Number,             // 游戏时长（秒）
  难度: String,             // easy/normal/hard
  createdAt: Date
}
```

**room_records（房间记录）**
```javascript
{
  _id: ObjectId,
  roomId: String,           // 房间唯一ID
  类型: String,             // 单机/人机/在线
  玩家数: Number,
  创建者: ObjectId,
  开始时间: Date,
  结束时间: Date,
  总底池: Number
}
```

### 3.2 Redis Keys

```
# 在线玩家状态
online:user:{userId} = {socketId, roomId, 进入时间}

# 匹配队列（Hash）
match:queue:normal = { userId1: timestamp, userId2: timestamp, ... }

# 房间信息
room:{roomId} = { 房间状态JSON }  (TTL: 2小时)

# 玩家金币缓存
cache:gold:{userId} = 12345  (TTL: 5分钟)
```

## 5. AI 引擎设计

### 5.1 手牌强度评估

```typescript
// 手牌强度 0-1
function evaluateHandStrength(cards: Card[]): number {
  const handType = getHandType(cards);  // 豹子/顺金/金花/顺子/对子/单张
  const pointValue = getPointValue(cards); // 点数值

  const typeScore = HAND_TYPE_SCORE[handType]; // 0.0-1.0
  const pointScore = pointValue / 13; // 归一化

  return typeScore * 0.7 + pointScore * 0.3;
}
```

### 5.2 AI 决策树

```typescript
function aiDecide(action: GameAction): AIResult {

  // 简单难度：主要看手牌强度
  if (difficulty === 'easy') {
```

## 6. Docker 部署配置

### 6.1 Dockerfile (服务器)

```dockerfile
FROM node:20-alpine

WORKDIR /app

# 安装依赖
COPY package*.json ./
RUN npm install --production

# 复制代码
COPY . .

# 暴露端口
EXPOSE 3000 8443

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s \
  CMD wget --no-verbose --tries=1 --spider http://localhost:3000/health || exit 1

CMD ["node", "server.js"]
```

### 6.2 docker-compose.yml

```yaml
version: '3.8'

services:
  game-server:
    build: .
    ports:
      - "3000:3000"   # HTTP API
      - "8443:8443"   # WebSocket
    environment:
      - NODE_ENV=production
      - PORT=3000
      - WS_PORT=8443
      - MONGODB_URI=mongodb://mongo:27017/zhajinhua
      - REDIS_HOST=redis
      - REDIS_PORT=6379
    depends_on:
      - mongo
      - redis
    restart: unless-stopped
    networks:
      - game-network

  mongo:
    image: mongo:6
    volumes:
      - mongo_data:/data/db
    restart: unless-stopped
    networks:
      - game-network

  redis:
    image: redis:7-alpine
    volumes:
      - redis_data:/data
    restart: unless-stopped
    networks:
      - game-network

volumes:
  mongo_data:
  redis_data:

networks:
  game-network:
    driver: bridge
```

### 6.3 nginx.conf (WebSocket反向代理)

```nginx
upstream game_ws {
    server game-server:8443;
}

server {
    listen 443 ssl;
    server_name your-game-domain.com;

    ssl_certificate /etc/letsencrypt/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/privkey.pem;

    location / {
        proxy_pass http://game_ws;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_read_timeout 86400;
    }
}
```
